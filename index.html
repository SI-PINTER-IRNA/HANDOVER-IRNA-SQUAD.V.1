Panduan Integrasi Google Spreadsheet

Ikuti langkah-langkah di bawah ini untuk menghubungkan aplikasi HANDOVER IRNA SQUAD dengan Google Spreadsheet Anda:

  1. Buat Spreadsheet Baru di Google Drive Anda.
  2. Di Google Sheet, buka menu Ekstensi (Extensions) > pilih Apps Script.
  3. Hapus seluruh kode default di editor Apps Script, lalu tempel kode berikut: 
    function doPost(e) {
  try {
    var json = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    if (sheet.getLastRow() === 0) {
      var headers = ["ID Pasien", "Tanggal Handover", "Nama Pasien", "No. Bed", "Diagnosa", "DPJP", "Dokter Lainnya", "Perawat PJ", "Shift", "Situation (S)", "Background (B)", "Assessment (A)", "Recommendation (R)", "Status", "Logs (JSON)"];
      sheet.appendRow(headers);
      var headerRange = sheet.getRange(1, 1, 1, headers.length);
      headerRange.setFontWeight("bold").setBackground("#1e3a8a").setFontColor("#ffffff").setHorizontalAlignment("center").setVerticalAlignment("middle");
      sheet.setRowHeight(1, 30);
    } else {
      var lastRow = sheet.getLastRow();
      if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).clearContent(); }
    }
    json.forEach(function(p) {
      sheet.appendRow([p.id, p.timestamp || new Date().toISOString(), p.name, p.bed, p.diagnosis, p.dpjp || '-', p.otherDoctors || '-', p.pj || '-', p.shift, p.s || '-', p.b || '-', p.a || '-', p.r || '-', p.status || 'Aktif', JSON.stringify(p.logs || [])]);
    });
    for (var col = 1; col <= 15; col++) { sheet.autoResizeColumn(col); }
    return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Berhasil menyelaraskan data!"})).setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    var lastRow = sheet.getLastRow();
    if (lastRow <= 1) { return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON); }
    var values = sheet.getRange(2, 1, lastRow - 1, 15).getValues();
    var data = [];
    for (var i = 0; i < values.length; i++) {
      var row = values[i];
      var logsArr = [];
      try { logsArr = JSON.parse(row[14]); } catch(e) { logsArr = []; }
      data.push({
        id: Number(row[0]), timestamp: row[1], name: row[2], bed: row[3], diagnosis: row[4],
        dpjp: row[5] === '-' ? '' : row[5], otherDoctors: row[6] === '-' ? '' : row[6],
        pj: row[7] === '-' ? '' : row[7], shift: row[8], s: row[9] === '-' ? '' : row[9],
        b: row[10] === '-' ? '' : row[10], a: row[11] === '-' ? '' : row[11], r: row[12] === '-' ? '' : row[12],
        status: row[13], logs: logsArr
      });
    }
    return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
  4. Klik ikon **Save (Simpan)** di Apps Script.
  5. Klik tombol **Terapkan (Deploy)** di kanan atas > pilih **Penerapan Baru (New deployment)**.
  6. Pilih jenis penerapan: **Aplikasi Web (Web app)**.
  7. Atur konfigurasi:
    • Jalankan sebagai (Execute as): Saya (Me)
    • Siapa yang memiliki akses (Who has access): Siapa saja (Anyone)
  8. Klik **Terapkan (Deploy)**. Jika diminta, klik **Berikan akses (Authorize access)** dan izinkan menggunakan akun Google Anda.
  9. Salin **URL Aplikasi Web** yang diberikan oleh Google.
  10. Tempelkan (*paste*) URL tersebut ke kolom input di sudut kanan atas aplikasi ini dan klik **Kirim Data**.
